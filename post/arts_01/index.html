题目

  • 把数字转换成罗马数字

就是把数字变罗马数字。 一开始看规则好像有点复杂，后来灵光一现，其实就是把所有规则列出来一个一个拼进去就好了。

var romanMap = map[int]string{
	1000: "M",
	900: "CM",
	500: "D",
	400: "CD",
	100: "C",
	90: "XC",
	50: "L",
	40: "XL",
	10: "X",
	9: "IX",
	5: "V",
	4: "IV",
	1: "I",
}

var nums = [13]int{1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1}

func intToRoman(num int) string {
	var sb strings.Builder
	for _, v := range nums{
		str := romanMap[v]
		for {
			if num - v >= 0{
				num = num - v
				sb.WriteString(str)
			} else{
				break
			}
		}
	}
	return sb.String()
}
